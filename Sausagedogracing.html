        // Game initialization

        document.addEventListener('DOMContentLoaded', function() {

            const canvas = document.getElementById('gameCanvas');

            const ctx = canvas.getContext('2d');

            const startBtn = document.getElementById('startBtn');

           

            // Set canvas size

            function setCanvasSize() {

                const container = canvas.parentElement;

                canvas.width = container.clientWidth;

                canvas.height = container.clientHeight;

            }

           

            setCanvasSize();

            window.addEventListener('resize', setCanvasSize);

           

            // Game variables

            let gameState = {

                running: false,

                currentLap: 1,

                maxLaps: 3,

                score: 0,

                difficulty: 'easy'

            };

           

            // Dog properties

            const dog = {

                x: canvas.width / 2,

                y: canvas.height - 100,

                width: 50,

                height: 25,

                speed: 0,

                maxSpeed: 8,

                acceleration: 0.15,

                deceleration: 0.08,

                moveSpeed: 4,

                color: '#8B4513',

                earColor: '#A0522D',

                noseColor: '#000000',

                eyeColor: '#000000',

                tongueColor: '#FF69B4',

                tailWag: 0,

                legMovement: 0,

                facingRight: true

            };

           

            // Track properties

            const track = {

                centerX: canvas.width / 2,

                centerY: canvas.height / 2,

                outerRadius: Math.min(canvas.width, canvas.height) * 0.4,

                innerRadius: Math.min(canvas.width, canvas.height) * 0.2

            };

           

            // Game elements

            let bones = [];

            let obstacles = [];

            let trees = [];

           

            // Input state

            const keys = {

                up: false,

                down: false,

                left: false,

                right: false

            };

           

            // Initialize game

            function initGame() {

                dog.x = canvas.width / 2;

                dog.y = canvas.height - 100;

                dog.speed = 0;

                dog.facingRight = true;

               

                bones = [];

                obstacles = [];

                trees = [];

               

                // Create bones

                for (let i = 0; i < 12; i++) {

                    const angle = Math.random() * Math.PI * 2;

                    const radius = track.innerRadius + 30 + Math.random() * (track.outerRadius - track.innerRadius - 60);

                    bones.push({

                        x: track.centerX + Math.cos(angle) * radius,

                        y: track.centerY + Math.sin(angle) * radius,

                        collected: false,

                        rotation: Math.random() * Math.PI * 2

                    });

                }

               

                // Create trees

                for (let i = 0; i < 20; i++) {

                    const angle = Math.random() * Math.PI * 2;

                    const radius = track.outerRadius + 40 + Math.random() * 100;

                    trees.push({

                        x: track.centerX + Math.cos(angle) * radius,

                        y: track.centerY + Math.sin(angle) * radius,

                        size: 25 + Math.random() * 20

                    });

                }

               

                adjustDifficulty();

            }

           

            function adjustDifficulty() {

                switch(gameState.difficulty) {

                    case 'easy':

                        dog.maxSpeed = 8;

                        dog.acceleration = 0.15;

                        obstacles = [];

                        break;

                    case 'medium':

                        dog.maxSpeed = 7;

                        dog.acceleration = 0.12;

                        for (let i = 0; i < 6; i++) {

                            const angle = Math.random() * Math.PI * 2;

                            const radius = track.innerRadius + 40 + Math.random() * (track.outerRadius - track.innerRadius - 80);

                            obstacles.push({

                                x: track.centerX + Math.cos(angle) * radius,

                                y: track.centerY + Math.sin(angle) * radius,

                                width: 25,

                                height: 25

                            });

                        }

                        break;

                    case 'hard':

                        dog.maxSpeed = 6;

                        dog.acceleration = 0.1;

                        for (let i = 0; i < 12; i++) {

                            const angle = Math.random() * Math.PI * 2;

                            const radius = track.innerRadius + 40 + Math.random() * (track.outerRadius - track.innerRadius - 80);

                            obstacles.push({

                                x: track.centerX + Math.cos(angle) * radius,

                                y: track.centerY + Math.sin(angle) * radius,

                                width: 25,

                                height: 25

                            });

                        }

                        break;

                }

            }

           

            function drawTrack() {

                ctx.fillStyle = '#7CFC00';

                ctx.fillRect(0, 0, canvas.width, canvas.height);

               

                ctx.fillStyle = '#708090';

                ctx.beginPath();

                ctx.arc(track.centerX, track.centerY, track.outerRadius, 0, Math.PI * 2);

                ctx.arc(track.centerX, track.centerY, track.innerRadius, 0, Math.PI * 2, true);

                ctx.fill();

               

                ctx.strokeStyle = '#FFFFFF';

                ctx.lineWidth = 3;

                ctx.setLineDash([25, 20]);

                ctx.beginPath();

                ctx.setLineDash([]);

                for (let i = 0; i < 10; i++) {

                    if (i % 2 === 0) continue;

            }

            function drawDog() {

                ctx.save();

                ctx.translate(dog.x, dog.y);

                if (!dog.facingRight) ctx.scale(-1, 1);

                if (Math.abs(dog.speed) > 0.5) ctx.translate(0, Math.sin(dog.legMovement) * 2);

               

                ctx.fillStyle = dog.color;

                ctx.fillRect(-dog.width/2, -dog.height/2, dog.width, dog.height);

               

                ctx.fillStyle = dog.color;

                ctx.beginPath();

                ctx.arc(dog.width/2, 0, dog.height/2, -Math.PI/2, Math.PI/2);

                ctx.fill();

               

                ctx.fillStyle = '#F5DEB3';

                ctx.beginPath();

                ctx.arc(dog.width/2 + 6, 0, 5, -Math.PI/2, Math.PI/2);

                ctx.fill();

               

                ctx.fillStyle = dog.noseColor;

                ctx.beginPath();

                ctx.arc(dog.width/2 + 9, 0, 3, 0, Math.PI * 2);

                ctx.fill();

               

                ctx.fillStyle = dog.eyeColor;

                ctx.beginPath();

                ctx.arc(dog.width/2 - 3, -5, 2.5, 0, Math.PI * 2);

                ctx.arc(dog.width/2 - 3, 5, 2.5, 0, Math.PI * 2);

                ctx.fill();

               

                ctx.fillStyle = 'white';

                ctx.beginPath();

                ctx.arc(dog.width/2 - 4, -6, 1, 0, Math.PI * 2);

                ctx.arc(dog.width/2 - 4, 4, 1, 0, Math.PI * 2);

                ctx.fill();

               

                ctx.fillStyle = dog.earColor;

                ctx.beginPath();

                    ctx.fill();

                }

               

                const tailAngle = Math.sin(dog.tailWag) * 0.5;

                ctx.rotate(tailAngle);

                ctx.fillStyle = dog.color;

                ctx.fill();

                ctx.restore();

            }

           

            function updateGame() {

                if (!gameState.running) return;

               

                dog.tailWag += 0.2;

                if (Math.abs(dog.speed) > 0.3) dog.legMovement += 0.4;

               

                if (keys.up) {

                    dog.speed += dog.acceleration;

                } else if (keys.down) {

                    dog.speed -= dog.acceleration * 1.3;

                } else {

                    if (dog.speed > 0) dog.speed = Math.max(0, dog.speed - dog.deceleration);

                    else if (dog.speed < 0) dog.speed = Math.min(0, dog.speed + dog.deceleration);

                }

               

                dog.speed = Math.max(-dog.maxSpeed/2, Math.min(dog.speed, dog.maxSpeed));

               

                if (keys.left) {

                    dog.x -= dog.moveSpeed;

                    dog.facingRight = false;

                }

                if (keys.right) {

                    dog.x += dog.moveSpeed;

                    dog.facingRight = true;

                }

               

                dog.y -= dog.speed;

                dog.x = Math.max(30, Math.min(dog.x, canvas.width - 30));

                dog.y = Math.max(30, Math.min(dog.y, canvas.height - 30));

               

                bones.forEach(bone => {

                    if (!bone.collected) {

                        const dx = dog.x - bone.x;

                        const dy = dog.y - bone.y;

                        const distance = Math.sqrt(dx * dx + dy * dy);

                        if (distance < 25) {

                            bone.collected = true;

                            gameState.score += 100;

                            updateUI();

                        }

                    }

                });

               

                obstacles.forEach(obstacle => {

                    const dx = Math.abs(dog.x - obstacle.x);

                    const dy = Math.abs(dog.y - obstacle.y);

                    if (dx < 25 && dy < 25) dog.speed *= 0.5;

                });

               

                const dx = dog.x - track.centerX;

                const dy = dog.y - track.centerY;

                const distance = Math.sqrt(dx * dx + dy * dy);

               

                if (dog.y < track.centerY - track.outerRadius + 20 &&

                    dog.x > track.centerX - 60 && dog.x < track.centerX + 60) {

                    if (gameState.currentLap < gameState.maxLaps) {

                        gameState.currentLap++;

                        updateUI();

                    } else {

                        gameState.running = false;

                        document.getElementById('statusMessage').textContent = `🏆 RACE COMPLETE! Score: ${gameState.score}`;

                    }

                }

               

                updateUI();

            }

           

            function updateUI() {

                document.getElementById('speed').textContent = Math.abs(Math.round(dog.speed * 20));

                document.getElementById('lap').textContent = `${gameState.currentLap}/${gameState.maxLaps}`;

                document.getElementById('score').textContent = gameState.score;

            }

                ctx.clearRect(0, 0, canvas.width, canvas.height);

                drawObstacles();

                updateGame();

                requestAnimationFrame(gameLoop);

            }

            function setupTouchButton(buttonId, key) {

                const button = document.getElementById(buttonId);

                button.addEventListener('mousedown', () => keys[key] = true);

                button.addEventListener('mouseup', () => keys[key] = false);

                button.addEventListener('mouseleave', () => keys[key] = false);

                button.addEventListener('touchstart', (e) => { e.preventDefault(); keys[key] = true; });

                button.addEventListener('touchend', (e) => { e.preventDefault(); keys[key] = false; });

            }

           

            setupTouchButton('upBtn', 'up');

            setupTouchButton('downBtn', 'down');

            setupTouchButton('leftBtn', 'left');

            setupTouchButton('rightBtn', 'right');

           

            window.addEventListener('keydown', (e) => {

                switch(e.key) {

                    case 'ArrowUp': keys.up = true; e.preventDefault(); break;

                    case 'ArrowDown': keys.down = true; e.preventDefault(); break;

                    case 'ArrowLeft': keys.left = true; e.preventDefault(); break;

                    case 'ArrowRight': keys.right = true; e.preventDefault(); break;

                }

            });

           

            window.addEventListener('keyup', (e) => {

                switch(e.key) {

                    case 'ArrowUp': keys.up = false; break;

                    case 'ArrowDown': keys.down = false; break;

                    case 'ArrowLeft': keys.left = false; break;

                    case 'ArrowRight': keys.right = false; break;

                }

            });

        });
